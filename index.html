#define BIO_VERSION 0x0100 

#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <string.h>

typedef struct {
    char* name;
    char* description;
    int rooted;
    int androidver;
    char* os;
} Device;

typedef struct {
    char* nickname;// 0x0
    int prog_since;// 0x8
    uint32_t skills;// 0xC
    char* github;// 0x10

    char** active_projects;// 0x18
    Device** devices;// 0x20

    int (*print_bio)(void*);// 0x28
} DevProfile;

int print_bio(void* self) 
{
    char skills[256] = {0};
    char version[16] = {0};

    sprintf(version, "%d.%d", (BIO_VERSION >> 8), (BIO_VERSION & 0xFF));
    printf("welcome to my bio (v%s)!\n\n", version);

    DevProfile* b = (DevProfile*)self;
    printf("my nickname: %s\n", b->nickname);
    printf("my github: %s\n", b->github);
    printf("programming since %d\n\n", b->prog_since);

    printf("my current projects:\n");
    for (int i = 0; b->active_projects[i] != NULL; ++i) {
        printf(" %s\n", b->active_projects[i]);
    }
    uint32_t mask = b->skills;

    if (mask & 0x1) strcat(skills, "C / ");
    if (mask & 0x2) strcat(skills, "C++ / ");
    if (mask & 0x4) strcat(skills, "C# / ");
    if (mask & 0x8) strcat(skills, "Android Reverse Engineering (prefer il2cpp games)");

    printf("\nmy skills in programming:\n   %s.", skills);

    printf("\n\nmy devices:\n");
    for (Device** d = b->devices; *d; d++) {
        Device* dev = *d;
        char v[16] = {0}; 
        if (dev->androidver != -1) sprintf(v, "Android %d ", dev->androidver);

        printf(" [%ld] %s (%s)\n   OS: %s%s\n   root: %s\n\n", 
            d - b->devices, dev->name, dev->description,
            v, dev->os,
            (dev->rooted == 0 ? "true" : "false"));
    }

    return 0;
}
